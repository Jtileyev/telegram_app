```python
def get_apartments(city_id=None, district_id=None, limit=10, offset=0):
    query = """
        SELECT
            a.*,
            GROUP_CONCAT(ap.photo_path, '|') as photo_paths,
            ap_main.photo_path as main_photo
        FROM apartments a
        LEFT JOIN apartment_photos ap ON a.id = ap.apartment_id
        LEFT JOIN apartment_photos ap_main ON a.id = ap_main.apartment_id
            AND ap_main.is_main = 1
        WHERE a.is_active = 1
    """

    params = []
    if city_id:
        query += " AND a.city_id = ?"
        params.append(city_id)

    if district_id and district_id != 'all':
        query += " AND a.district_id = ?"
        params.append(district_id)

    query += " GROUP BY a.id ORDER BY a.created_at DESC LIMIT ? OFFSET ?"
    params.extend([limit, offset])

    conn = get_connection()
    apartments = conn.execute(query, params).fetchall()

    # Разбить photos строку в массив
    result = []
    for apt in apartments:
        apt_dict = dict(apt)
        apt_dict['photos'] = apt_dict['photo_paths'].split('|') if apt_dict['photo_paths'] else []
        del apt_dict['photo_paths']
        result.append(apt_dict)

    conn.close()
    return result
```

- [ ] **Действие:** Создать:
```python
import sqlite3
from pathlib import Path
from typing import Optional

class DatabaseManager:
    def __init__(self, db_path: str):
        self.db_path = db_path
        self._connection: Optional[sqlite3.Connection] = None

    def __enter__(self):
        self._connection = sqlite3.connect(self.db_path)
        self._connection.row_factory = sqlite3.Row
        return self._connection

    def __exit__(self, exc_type, exc_val, exc_tb):
        if self._connection:
            if exc_type is None:
                self._connection.commit()
            else:
                self._connection.rollback()
            self._connection.close()
        return False
```
